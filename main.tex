% total number of tabbed chapters (controls vertical spread of the tabs)
\newcommand{\TotalTabChapters}{6}
\newlength{\tabwidth}\setlength{\tabwidth}{0.30in}

\newlength{\tabtop}\setlength{\tabtop}{1.35in}   % top of the first tab, from page top
\newlength{\tabstep}\setlength{\tabstep}{0.97in} % top-to-top spacing between tabs
\newlength{\taboffset} % scratch: distance from page top to this tab's top edge

% draw one thumb tab on the outer edge, vertically offset by chapter number.
% right edge on odd (recto) pages, left edge on even (verso) pages.
\newcommand{\drawthumbtab}{%
  \ifnum\value{chapter}>0
    \setlength{\taboffset}{\tabtop + \tabstep*(\value{chapter}-1)}%
    \begin{tikzpicture}[remember picture, overlay]
      \ifodd\value{page}% recto: outer edge is on the right
        \fill[accent]
          ([yshift=-\taboffset]current page.north east)
          rectangle ++(-\tabwidth,-\tabheight);
        \node[white, font=\sffamily\bfseries\large]
          at ([xshift=-0.5\tabwidth, yshift=-\taboffset-0.5\tabheight]current page.north east)
          {\thechapter};
      \else% verso: outer edge is on the left
        \fill[accent]
          ([yshift=-\taboffset]current page.north west)
          rectangle ++(\tabwidth,-\tabheight);
        \node[white, font=\sffamily\bfseries\large]
          at ([xshift=0.5\tabwidth, yshift=-\taboffset-0.5\tabheight]current page.north west)
          {\thechapter};
      \fi
    \end{tikzpicture}%
  \fi
}
\AddToShipoutPictureBG{\drawthumbtab}
